*** Tested on Ubuntu 16.04 (AWS T2 instance). ***

Overview:

Steps [1] and [2] need be done just once. Step [3] is done whenever you want to
update the Spigot server code to get bug/security fixes, new features, etc. Step
[4] is done whenever you want to create a new Spigot server instance.

Step-by-Step:

[1] Setup the host server OS using server-setup.txt.

[2] Setup the Forge environment:
    set minecraft_user=minecraft
    set minecraft_dir=/var/minecraft
    set forge_dir=${minecraft_dir}/forge
    sudo mkdir -p ${forge_dir}/{build,instances,shared}
    sudo ln -sf ${minecraft_dir}/shared/ops.json       ${forge_dir}/shared/ops.json        # symlink the server-wide list of ops
    sudo ln -sf ${minecraft_dir}/shared/whitelist.json ${forge_dir}/shared/whitelist.json  # symlink the server-wide list of whitelist players
    sudo chown -R ${minecraft_user}:${minecraft_user}  ${forge_dir}

[3] Build/update the Spigot server code:
    set minecraft_user=minecraft
    set minecraft_dir=/var/minecraft
    set forge_dir=${minecraft_dir}/forge
    cd ${forge_dir}/build
    # TODO

[4] Create a new Spigot server instance:
    set minecraft_user=minecraft
    set forge_dir=/var/minecraft/forge
    set instance_name=test-instance                                                       # name the instance - in this example it's named 'main'
    set instance_dir=${forge_dir}/instances/${instance_name}                              # everything belonging to this instance is in this dir
    sudo mkdir -p ${instance_dir}/worlds                                                  # create the instance directory and the worlds sub-directory
    sudo ln -sf ${forge_dir}/build/spigot*.jar        ${instance_dir}/spigot.jar         # symlink the spigot jar, select a specific revision if there are multiple
    sudo ln -sf ${forge_dir}/build/craftbukkit*.jar   ${instance_dir}/craftbukkit.jar    # symlink the craftbukkit jar, select a specific revision if there are multiple
    sudo ln -sf ${forge_dir}/shared/ops.json          ${instance_dir}/ops.json           # symlink the list of ops
    sudo ln -sf ${forge_dir}/shared/whitelist.json    ${instance_dir}/whitelist.json     # symlink the list of whitelist players
    sudo ln -sf ${forge_dir}/shared/server.properties ${instance_dir}/server.properties  # symlink the server properties file
    sudo ln -sf ${forge_dir}/shared/start.sh          ${instance_dir}/start.sh           # symlink the start script
    sudo chown -R ${minecraft_user}:${minecraft_user} ${instance_dir}
